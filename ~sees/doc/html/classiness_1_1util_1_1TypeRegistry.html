template<typename... T_SUPPORTED_TYPES>
class iness::util::TypeRegistry< T_SUPPORTED_TYPES >

Provides compile time type information (ids) on a fixed, configurable set of types and functionality to instantiate and call templated objects automatically with the registered types.
